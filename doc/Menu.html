def initialize(nombre, &block)
  @nombre = nombre
  @desc = ''
  @componentes = []
  @precio = 0.0

  return unless block_given?

  if block.arity == 1
    yield self
  else
    instance_eval(&block)
  end
end

def componente(args = {})

  @precio += args[:precio]
end

def to_s
  string = @nombre
  string << "\nComponentes:"
  @componentes.each do |componente|
    plato = componente[:plato]
    string << "\n\t"
    string << "#{plato.nombre} => #{componente[:precio]}€, #{plato.vct} kcal, #{plato.emisiones[:co2]} CO2"
  end
  string
end
