def initialize(data = nil)
  if data.nil?
    @head = @tail = nil
    @size = 0
  else
    @head = Node.new(data, nil, nil)
    @tail = @head
    @size = 1
  end
end

def [](index)
  return nil if index.negative? || index >= @size

  current_node = @head
  (0..index - 1).each { |_i| current_node = current_node.next }
  current_node.data
end

def insert(data, index)
  if index.zero?
    push_front(data)
    return
  end

  temp = @head

  (0..index - 2).each do |_i|
    break if temp.next.nil?

    temp = temp.next
  end

  next_node = temp.next
  temp.next = Node.new(data, temp, next_node)
  @tail = temp.next if next_node.nil?
  @size += 1
end

def pop_back
  return nil if @size.zero?

  if @size == 1
    temp = @head.data
    @head = @tail = nil
  else
    temp = @tail.data
    @tail.prev.next = nil
    @tail = @tail.prev
  end
  @size -= 1
  temp
end

            @tail = nil

  temp

def to_s
  result = ''
  current_node = @head

  until current_node.next.nil?
    result << current_node.data.to_s << ' | '
    current_node = current_node.next
  end
  result << current_node.data.to_s
  result
end
